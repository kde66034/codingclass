CSS 소스 : 반응형 버그 해결하기

반응형(모바일, 태블릿 PC 전용 화면) 버그를 해결하기 위해 아래와 같은 코드를 입력해주었다. 

<style>
    /* Windows에서 페이지 이동 시 웹뷰화면 흔들림 버그 없애줌 */
    body{
        overflow: overlay;
    }

    /* 햄버거 모양 메뉴 클릭 시 스크롤바를 없애줌 */
    body.fixed {
        overflow: hidden;
        height: 100vh;
    }
</style>

JS 소스

첫번째로는 스크롤을 이동할 때마다 현재 위치한 메뉴에 active 효과를 적용시키는 스크립트를 입력하였고, 두번째는 스크롤 이동을 부드럽게 하기 위해 behavior:"smooth"를 적용하였으며, 마지막으로는 반응형 전용 햄버거 모양 메뉴 버튼과 해당 버튼을 클릭(또는 터치)하면 메뉴 바를 나타나게 하고, X 버튼을 누르면 사라지게 하도록 스크립트를 짜 놓았다. 

<script>
    window.addEventListener("scroll", () => {
        let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop;

        document.querySelectorAll(".scroll").forEach((element, index) => {
            if(scrollTop >= element.offsetTop-500){
                document.querySelectorAll(".header__menu li").forEach(li => {
                    li.classList.remove("active");
                });
                document.querySelector(".header__menu li:nth-child("+(index+1)+")").classList.add("active");
            };
        });
    });

    // 스크롤 이동
    document.querySelectorAll("#headerType .header__inner .header__menu li a").forEach(li => {
        li.addEventListener("click", (e) => {
            e.preventDefault();
            document.querySelector(li.getAttribute("href")).scrollIntoView({
                behavior: "smooth"  // 이동 부드럽게
            });
        });
    });

    // 메뉴 버튼 (반응형)
    const btnHam = document.querySelector(".header__ham");
    const btnMenu = document.querySelector(".header__menu");
    const btnMenuList = btnMenu.querySelectorAll(".header__menu ul li a");

    btnHam.addEventListener("click", () => {
        btnHam.classList.toggle("active");
        btnMenu.classList.toggle("active");
        document.body.classList.toggle("fixed");
    });

    // 반응형에서 메뉴 버튼 누르는 즉시 메뉴창 닫히게 하기

            document.body.classList.remove("fixed");
