JavaScript 소스 보기

검색 이펙트 05의 JavaScript 소스와 부연 설명(주석 처리됨)이다. 상수 cssProperty에 검색결과에 띄울 CSS 속성 리스트들을 모두 저장하고 이것들을 출력하고 속성 클릭 시 설명이 나오도록 하는 작업과 마지막으로 사용자가 입력한 값과 일치하는 속성 name의 desc를 출력시켜주는 스크립트를 짠다. 숫자를 별로 치환하는 작업은 아래 주석 처리된 부연 설명을 참조할 것. 

// CSS 속성 리스트 저장
const cssProperty = [
    { name: "accent-color", desc: "특정 요소에 색상을 지정할 때 사용됩니다." },
    { name: "align-content", desc: "콘텐츠 아이템의 상하관계 정렬 상태를 설정합니다." },
    { name: "align-items", desc: "콘텐츠 아이템의 내부 상하관계 정렬 상태를 설정합니다." },
]

const searchList = document.querySelector(".search__list ul");
const searchClick = document.querySelectorAll(".search__click ul li");
const star = document.querySelectorAll(".search__click ul li a")
const searchNum = document.querySelector(".search__info .num");        // 전체 개수

let count=0;

// 출력하기
function updateList(list){
    listCSS = "";

    for(const data of list){
        starList = star[data.star].innerText;   // 숫자를 별로 치환함
        count++;

        listCSS += `<li><strong>${data.name}</strong> : ${data.desc} <em>${starList}</em></li>`;
        // 속성 이름 : 설명 <em>숫자를 별로 치환한 값</em>
    }
    searchNum.innerHTML = count;
    count = 0;

    searchList.innerHTML = listCSS;
}
updateList(cssProperty);

// 클릭하기
searchClick.forEach(star => {
    star.addEventListener("click", () => {
        const target = star.dataset.star;
        const target2 = star.dataset.number;

        const filterList = cssProperty.filter(data => data.star == target || data.star >= target2);
        
        searchNum.innerHTML = filterList.length;        // 속성 개수 표시

        updateList(filterList);
    });
});
