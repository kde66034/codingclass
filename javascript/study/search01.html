JavaScript 소스 보기

검색 이펙트 01의 JavaScript 소스와 부연 설명(주석 처리됨)이다. 

// 선택자 : 상수 const를 필요한 만큼 선언 후 HTML 영역의 div, 시맨틱태그 등의 class명과 연결
const searchBox = document.querySelector(".search__box input");   // 검색 영역
const searchList = document.querySelectorAll(".search__list ul li");    // 목록 리스트

// 검색 영역 이벤트
searchBox.addEventListener("keyup", () => { // 사용자가 검색 영역에 keyup 했을 때(키를 눌렀다가 떼었을 때)
    const searchWord = searchBox.value;     // 사용자가 입력한 키워드 값
            
    searchList.forEach(el => {
        const htmlName = el.dataset.name;   // 데이터 셋팅

        if(htmlName.indexOf(searchWord)){
            el.classList.add("hide");       // 사용자가 입력한 키워드의 알파벳이 포함된 내용 출력
        } else {
            el.classList.remove("hide");    // indexOf값이 0이 되는 값만 hide를 지워줌
        }
    });
});
