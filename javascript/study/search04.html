JavaScript 소스 보기

검색 이펙트 04의 JavaScript 소스와 부연 설명(주석 처리됨)이다. 상수 cssProperty에 검색결과에 띄울 CSS 속성 리스트들을 모두 저장하고 이것들을 출력하고 속성 클릭 시 설명이 나오도록 하는 작업과 마지막으로 사용자가 입력한 값과 일치하는 속성 name의 desc를 출력시켜주는 스크립트를 짠다. 

// CSS 속성 리스트 저장
const cssProperty = [
    { name: "accent-color", desc: "특정 요소에 색상을 지정할 때 사용됩니다." },
    { name: "align-content", desc: "콘텐츠 아이템의 상하관계 정렬 상태를 설정합니다." },
    { name: "align-items", desc: "콘텐츠 아이템의 내부 상하관계 정렬 상태를 설정합니다." },
]

const searchList = document.querySelector(".search__list");         // 서치 리스트 ul
const searchNum = document.querySelector(".search__info .num");        // 전체 개수
const searchBox = document.querySelector(".search__box input");        // 검색창
const searchDesc = document.querySelector(".search__desc");            // 설명창

// 출력하기
cssProperty.map((element, index) => {
    searchNum.innerHTML = cssProperty.length; // CSS 속성 총 개수 출력
    // searchNum.innerHTML = index + 1;

    searchList.innerHTML += `${element.name}`;
});

// 속성 클릭 작업
const searchListSpan = searchList.querySelectorAll('span');
searchListSpan.forEach((span) => {
    span.addEventListener('click', (e) => {
        const seachWord = e.target.textContent;
        findProperty(seachWord);
        window.scrollTo({left : 0, top : 0, behavior : "smooth"});
    });
});

// 입력하기
searchBox.addEventListener("keyup", () => {
    const searchWord = searchBox.value;     // 사용자가 입력한 값을 저장
    // console.log(searchWord);

    findProperty(searchWord);
});

// 속성 찾기
function findProperty(searchData) {
    const targetData = cssProperty.find((data) => data.name === searchData);

    if(targetData == null) {    // 데이터가 null일(존재하지 않을) 경우
        searchDesc.textContent = "해당 속성은 존재하지 않습니다. 다시 검색해주세요!";   // 데이터가 존재하지 않을 시 표시할 문구
        return;     // 종료의 의미 - F12 콘솔창 오류를 없애줌
    };

    searchDesc.textContent = targetData.desc;
};
