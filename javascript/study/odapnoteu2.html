    }

}

10. 다음의 결괏값을 작성하시오.

{
    function func(s){
        let answer = "";
        for(let x of s){        // x 변수에 "CodingNote" 문자열 저장
            if( x !== 'C') answer += '+';    // x 중 'C'가 아닌 것들을 '+'으로 바꾼다
            else answer += x;       // 아니면 그대로 두기?
        }
        return answer;              // "codingNote" 중 'C'가 아닌 걸 전부 +으로 바꿈
                                    // 정답 == ++++++++++
    }
    
    let str = "codingNote";
    document.write(func(str));
}
