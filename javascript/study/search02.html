JavaScript 소스 보기

검색 이펙트 02의 JavaScript 소스와 부연 설명(주석 처리됨)이다. 

// 선택자 : 상수 const를 필요한 만큼 선언 후 HTML 영역의 div, 시맨틱태그 등의 class명과 연결
const searchBox = document.querySelector(".search__box input");   // 검색 영역
const searchList = document.querySelectorAll(".search__list ul li");    // 목록 리스트
const searchInfo = document.querySelector(".search__info .num");        // 전체 개수

// 전체 개수 구하기
searchInfo.textContent = searchList.length; // CSS 속성 총 개수 출력

// 전체 개수 구하기 : for문 (복잡, 권장하지 않음)
// for(let i = 0; i <= searchList.length; i++) {
//     const searchInfo = document.querySelector(".search__info .num");
    
//     searchInfo.innerText = i;
// }

// 검색영역
searchBox.addEventListener("keyup", () => {
    const searchWord = searchBox.value;     // 사용자가 입력한 키워드

    searchList.forEach(el => {
        const cssName = el.dataset.name;    // CSS 속성 모든 값

        if(cssName.includes(searchWord)) {
            el.classList.remove("hide");   // 일치값이 있으면 보여줌(숨김 효과를 제거)
        } else {
            el.classList.add("hide");    // 없으면 사라지게 하기(숨김 효과 활성화)
        }
    });
});
