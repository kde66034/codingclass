패럴럭스 이펙트 06

이번 시간에는 그 동안 배웠던 JavaScript를 이용하여 패럴럭스(parallax, 패럴랙스) 이펙트를 만들어보도록 하겠습니다. 먼저 parallax란, '시차(視差)'라는 의미를 가진 천문학 용어이며, 이번 포스팅에서 내비게이션 바에서 특정 섹션으로 가는 버튼을 누르면 해당 섹션으로 시차를 두고 스무스하게 이동시켜 주는 효과를 만들어보고자 합니다.

이번 패럴럭스 이펙트 06 - 텍스트 효과에서는 텍스트가 타이핑하는 형식으로 한글자씩 나타나는 애니메이션을 주겠습니다.

    //     splitTextWrap = "<span>" + splitTextWrap + "</span>";
    //     text.innerHTML = splitTextWrap;
    // });

    document.querySelectorAll('.split').forEach((text) => {
        let splitText = text.innerText;
        let splitWrap = splitText.split('').join('</span><span aria-hidden = "true">');
        splitWrap = "<span aria-hidden = 'true'>" + splitWrap + "</span>";
        text.innerHTML = splitWrap;
        text.setAttribute("aria-label", splitText);
    });

    function scroll() {
        let scrollTop = window.scrollY;
        document.querySelector('.scroll span').innerText = Math.ceil(scrollTop);

        // css 클래스 추가(수동적 방법)
        // document.querySelectorAll('.content__item').forEach((item) => {
        //     if(scrollTop >= item.offsetTop - 200) {
        //         item.querySelector('.split').classList.add('show');
        //     }
        // });
        
        document.querySelectorAll('.content__item').forEach((item) => {
            if(scrollTop >= item.offsetTop - 200) {
                item.querySelector('.split').classList.add('show');
                item.querySelectorAll('.split span').forEach((span, index) => {
                    span.style.transitionDelay = `${index * 50}ms`;
                });
            }
        });
        requestAnimationFrame(scroll);
    }
    scroll();

    // 첫 번째 span 0.01, 
    // 두 번째 span 0.02, 
    // 세 번째 span 0.03
</script>
